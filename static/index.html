BROWSERIFY JUST GOT MORE MAGICAL.

Example: This one colors its own listings.

    <script src='http://browserify.nodejitsu.com/?require=["mrcolor"]' type="text/javascript"></script>
    <script type="text/javascript">
        window.onload = function () {
            var mr = require("mrcolor");
            var next = mr();

            var example = document.getElementById("example");

            example.innerHTML = example.innerHTML
                .split("\n")
                .map(function (line) {

                    var color = next().rgb();

                    var rgbText = "rgb(" + color.join(",") + ");";

                    return "<span style=\"color: " + rgbText + ";\">"
                        + line
                        + "</span>";
                }).join("\n");
        };
    </script>
