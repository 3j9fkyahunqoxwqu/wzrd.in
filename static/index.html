BROWSERIFY JUST GOT MORE MAGICAL.

Example: See those colors?^^ This is where they came from.

    <script src='http://browserify.nodejitsu.com/?require=["mrcolor", {"jquery": "jquery-browserify"}]'  type="text/javascript"></script>
    <script type="text/javascript">
      var color = require("mrcolor")(),
          $ = require("jquery");

      $(function () {
        $(".rainbow").replaceWith((function () {
          var h = $("<h1>");
          $(".rainbow")
            .text()
            .split("")
            .forEach(function (c) {
              h.append($("<span>")
                .css("color", "rgb(" + color().rgb().join(",")+")")
                .text(c)
              );
            });

          return h.addClass("rainbow");
        })());
      });
    </script>
